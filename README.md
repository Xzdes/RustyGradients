    );

    // 2. –°–æ–∑–¥–∞–µ–º –º–æ–¥–µ–ª—å
    let model = Sequential::new(vec![
        Box::new(Linear::new(2, 4)),
        Box::new(ReLU::new()),
        Box::new(Linear::new(4, 1)),
    ]);

    // 3. –°–æ–∑–¥–∞–µ–º –æ–ø—Ç–∏–º–∏–∑–∞—Ç–æ—Ä
    let mut optimizer = Adam::new(model.parameters(), 0.01, None, None);

    // 4. –¶–∏–∫–ª –æ–±—É—á–µ–Ω–∏—è
    for epoch in 0..=1000 {
        let predictions = model.forward(&training_data)?;
        let loss = mse_loss(&predictions, &training_labels);
        loss.backward();
        optimizer.step();
        optimizer.zero_grad();
        if epoch % 100 == 0 {
            println!("–≠–ø–æ—Ö–∞: {}, –ü–æ—Ç–µ—Ä–∏: {:.4}", epoch, loss.data.borrow().sum());
        }
    }
    Ok(())
}
